116    particlesRef.current.forEach(particle => {

118        scale: 0,
119        opacity: 0,
120        duration: 0.3,
121        ease: 'back.in(1.7)',
122        onComplete: () => {

124        }
125      });
126    });
127    particlesRef.current = [];
128  }, []);

130  const animateParticles = useCallback(() => {
131    if (!cardRef.current || !isHoveredRef.current) return;
132
133    if (!particlesInitialized.current) {
134      initializeParticles();
135    }
136
137    memoizedParticles.current.forEach((particle, index) => {
138      const timeoutId = setTimeout(() => {
139        if (!isHoveredRef.current || !cardRef.current) return;

141        const clone = particle.cloneNode(true);
142        cardRef.current.appendChild(clone);
143        particlesRef.current.push(clone);

145        gsap.fromTo(clone, { scale: 0, opacity: 0 }, { scale: 1, opacity: 1, duration: 0.3, ease: 'back.out(1.7)' });

147        gsap.to(clone, {
148          x: (Math.random() - 0.5) * 100,

151          duration: 2 + Math.random() * 2,
152          ease: 'none',
153          repeat: -1,
154          yoyo: true

157        gsap.to(clone, {

162          yoyo: true

167    });

173    const element = cardRef.current;

187      }
188    };
189
190    const handleMouseLeave = () => {
191      isHoveredRef.current = false;
192      clearAllParticles();
193
194      if (enableTilt) {
195        gsap.to(element, {
196          rotateX: 0,
197          rotateY: 0,
198          duration: 0.3,
199          ease: 'power2.out'

201      }
202
203      if (enableMagnetism) {
204        gsap.to(element, {
205          x: 0,
206          y: 0,
207          duration: 0.3,
208          ease: 'power2.out'
209        });
210      }
211    };
212
213    const handleMouseMove = e => {
214      if (!enableTilt && !enableMagnetism) return;

216      const rect = element.getBoundingClientRect();

218      const y = e.clientY - rect.top;
219      const centerX = rect.width / 2;

228          rotateY,
229          duration: 0.1,
230          ease: 'power2.out',
231          transformPerspective: 1000
232        });
233      }
234
235      if (enableMagnetism) {
236        const magnetX = (x - centerX) * 0.05;
237        const magnetY = (y - centerY) * 0.05;

241          y: magnetY,
242          duration: 0.3,
243          ease: 'power2.out'

245      }
246    };

249      if (!clickEffect) return;

251      const rect = element.getBoundingClientRect();
252      const x = e.clientX - rect.left;
253      const y = e.clientY - rect.top;
254
255      const maxDistance = Math.max(
256        Math.hypot(x, y),
257        Math.hypot(x - rect.width, y),
258        Math.hypot(x, y - rect.height),
259        Math.hypot(x - rect.width, y - rect.height)
260      );
261

265        width: ${maxDistance * 2}px;

269        left: ${x - maxDistance}px;
270        top: ${y - maxDistance}px;
271        pointer-events: none;
272        z-index: 1000;
273      `;
274
275      element.appendChild(ripple);
276
277      gsap.fromTo(
278        ripple,
279        {
280          scale: 0,
281          opacity: 1
282        },
283        {
284          scale: 1,
285          opacity: 0,
286          duration: 0.8,
287          ease: 'power2.out',
288          onComplete: () => ripple.remove()
289        }
290      );
291    };
292
293    element.addEventListener('mouseenter', handleMouseEnter);
294    element.addEventListener('mouseleave', handleMouseLeave);
295    element.addEventListener('mousemove', handleMouseMove);
296    element.addEventListener('click', handleClick);
297
298    return () => {
299      isHoveredRef.current = false;
300      element.removeEventListener('mouseenter', handleMouseEnter);
301      element.removeEventListener('mouseleave', handleMouseLeave);
302      element.removeEventListener('mousemove', handleMouseMove);
303      element.removeEventListener('click', handleClick);
304      clearAllParticles();
305    };
306  }, [animateParticles, clearAllParticles, disableAnimations, enableTilt, enableMagnetism, clickEffect, glowColor]);

319const GlobalSpotlight = ({
320  gridRef,
321  disableAnimations = false,
322  enabled = true,
323  spotlightRadius = DEFAULT_SPOTLIGHT_RADIUS,
324  glowColor = DEFAULT_GLOW_COLOR
325}) => {
326  const spotlightRef = useRef(null);
327  const isInsideSection = useRef(false);
328
329  useEffect(() => {
330    if (disableAnimations || !gridRef?.current || !enabled) return;
331
332    const spotlight = document.createElement('div');
333    spotlight.className = 'global-spotlight';
334    spotlight.style.cssText = `
335      position: fixed;
336      width: 800px;
337      height: 800px;
338      border-radius: 50%;
339      pointer-events: none;
340      background: radial-gradient(circle,
341        rgba(${glowColor}, 0.15) 0%,
342        rgba(${glowColor}, 0.08) 15%,
343        rgba(${glowColor}, 0.04) 25%,
344        rgba(${glowColor}, 0.02) 40%,
345        rgba(${glowColor}, 0.01) 65%,
346        transparent 70%
347      );
348      z-index: 200;
349      opacity: 0;
350      transform: translate(-50%, -50%);
351      mix-blend-mode: screen;
352    `;
353    document.body.appendChild(spotlight);
354    spotlightRef.current = spotlight;
355
356    const handleMouseMove = e => {
357      if (!spotlightRef.current || !gridRef.current) return;
358
359      const section = gridRef.current.closest('.bento-section');
360      const rect = section?.getBoundingClientRect();
361      const mouseInside =
362        rect && e.clientX >= rect.left && e.clientX <= rect.right && e.clientY >= rect.top && e.clientY <= rect.bottom;
363
364      isInsideSection.current = mouseInside || false;
365      const cards = gridRef.current.querySelectorAll('.card');
366
367      if (!mouseInside) {
368        gsap.to(spotlightRef.current, {
369          opacity: 0,
370          duration: 0.3,
371          ease: 'power2.out'
372        });
373        cards.forEach(card => {
374          card.style.setProperty('--glow-intensity', '0');
375        });
376        return;
377      }
378
379      const { proximity, fadeDistance } = calculateSpotlightValues(spotlightRadius);
380      let minDistance = Infinity;
381
382      cards.forEach(card => {
383        const cardElement = card;
384        const cardRect = cardElement.getBoundingClientRect();
385        const centerX = cardRect.left + cardRect.width / 2;
386        const centerY = cardRect.top + cardRect.height / 2;
387        const distance =
388          Math.hypot(e.clientX - centerX, e.clientY - centerY) - Math.max(cardRect.width, cardRect.height) / 2;
389        const effectiveDistance = Math.max(0, distance);
390
391        minDistance = Math.min(minDistance, effectiveDistance);
392
393        let glowIntensity = 0;
394        if (effectiveDistance <= proximity) {
395          glowIntensity = 1;
396        } else if (effectiveDistance <= fadeDistance) {
397          glowIntensity = (fadeDistance - effectiveDistance) / (fadeDistance - proximity);
398        }
399
400        updateCardGlowProperties(cardElement, e.clientX, e.clientY, glowIntensity, spotlightRadius);
401      });
402
403      gsap.to(spotlightRef.current, {
404        left: e.clientX,
405        top: e.clientY,
406        duration: 0.1,
407        ease: 'power2.out'
408      });
409
410      const targetOpacity =
411        minDistance <= proximity
412          ? 0.8
413          : minDistance <= fadeDistance
414            ? ((fadeDistance - minDistance) / (fadeDistance - proximity)) * 0.8
415            : 0;
416
417      gsap.to(spotlightRef.current, {
418        opacity: targetOpacity,
419        duration: targetOpacity > 0 ? 0.2 : 0.5,
420        ease: 'power2.out'
421      });
422    };
423
424    const handleMouseLeave = () => {
425      isInsideSection.current = false;
426      gridRef.current?.querySelectorAll('.card').forEach(card => {
427        card.style.setProperty('--glow-intensity', '0');
428      });
429      if (spotlightRef.current) {
430        gsap.to(spotlightRef.current, {
431          opacity: 0,
432          duration: 0.3,
433          ease: 'power2.out'
434        });
435      }
436    };
437
438    document.addEventListener('mousemove', handleMouseMove);
439    document.addEventListener('mouseleave', handleMouseLeave);
440
441    return () => {
442      document.removeEventListener('mousemove', handleMouseMove);
443      document.removeEventListener('mouseleave', handleMouseLeave);
444      spotlightRef.current?.parentNode?.removeChild(spotlightRef.current);
445    };
446  }, [gridRef, disableAnimations, enabled, spotlightRadius, glowColor]);
447
448  return null;
449};

661                  enableMagnetism={enableMagnetism}
662                >
663                  <div className="card__header flex justify-between gap-3 relative text-white">
664                    <span className="card__label text-base">{card.label}</span>
665                  </div>
666                  <div className="card__content flex flex-col relative text-white">
667                    <h3 className={`card__title font-normal text-base m-0 mb-1 ${textAutoHide ? 'text-clamp-1' : ''}`}>
668                      {card.title}
669                    </h3>
670                    <p
671                      className={`card__description text-xs leading-5 opacity-90 ${textAutoHide ? 'text-clamp-2' : ''}`}
672                    >
673                      {card.description}
674                    </p>
675                  </div>
676                </ParticleCard>
677              );
678            }
679
680            return (
681              <div
682                key={index}
683                className={baseClassName}
684                style={cardStyle}
685                ref={el => {
686                  if (!el) return;
687
688                  const handleMouseMove = e => {
689                    if (shouldDisableAnimations) return;
690
691                    const rect = el.getBoundingClientRect();
692                    const x = e.clientX - rect.left;
693                    const y = e.clientY - rect.top;
694                    const centerX = rect.width / 2;
695                    const centerY = rect.height / 2;
696
697                    if (enableTilt) {
698                      const rotateX = ((y - centerY) / centerY) * -10;
699                      const rotateY = ((x - centerX) / centerX) * 10;
700
701                      gsap.to(el, {
702                        rotateX,
703                        rotateY,
704                        duration: 0.1,
705                        ease: 'power2.out',
706                        transformPerspective: 1000
707                      });
708                    }
709
710                    if (enableMagnetism) {
711                      const magnetX = (x - centerX) * 0.05;
712                      const magnetY = (y - centerY) * 0.05;
713
714                      gsap.to(el, {
715                        x: magnetX,
716                        y: magnetY,
717                        duration: 0.3,
718                        ease: 'power2.out'
719                      });
720                    }
721                  };
722
723                  const handleMouseLeave = () => {
724                    if (shouldDisableAnimations) return;
725
726                    if (enableTilt) {
727                      gsap.to(el, {
728                        rotateX: 0,
729                        rotateY: 0,
730                        duration: 0.3,
731                        ease: 'power2.out'
732                      });
733                    }
734
735                    if (enableMagnetism) {
736                      gsap.to(el, {
737                        x: 0,
738                        y: 0,
739                        duration: 0.3,
740                        ease: 'power2.out'
741                      });
742                    }
743                  };
744
745                  const handleClick = e => {
746                    if (!clickEffect || shouldDisableAnimations) return;
747
748                    const rect = el.getBoundingClientRect();
749                    const x = e.clientX - rect.left;
750                    const y = e.clientY - rect.top;
751
752                    const maxDistance = Math.max(
753                      Math.hypot(x, y),
754                      Math.hypot(x - rect.width, y),
755                      Math.hypot(x, y - rect.height),
756                      Math.hypot(x - rect.width, y - rect.height)
757                    );
758
759                    const ripple = document.createElement('div');
760                    ripple.style.cssText = `
761                      position: absolute;
762                      width: ${maxDistance * 2}px;
763                      height: ${maxDistance * 2}px;
764                      border-radius: 50%;
765                      background: radial-gradient(circle, rgba(${glowColor}, 0.4) 0%, rgba(${glowColor}, 0.2) 30%, transparent 70%);
766                      left: ${x - maxDistance}px;
767                      top: ${y - maxDistance}px;
768                      pointer-events: none;
769                      z-index: 1000;
770                    `;
771
772                    el.appendChild(ripple);
773
774                    gsap.fromTo(
775                      ripple,
776                      {
777                        scale: 0,
778                        opacity: 1
779                      },
780                      {
781                        scale: 1,
782                        opacity: 0,
783                        duration: 0.8,
784                        ease: 'power2.out',
785                        onComplete: () => ripple.remove()
786                      }
787                    );
788                  };
789
790                  el.addEventListener('mousemove', handleMouseMove);
791                  el.addEventListener('mouseleave', handleMouseLeave);
792                  el.addEventListener('click', handleClick);
793                }}
794              >
795                <div className="card__header flex justify-between gap-3 relative text-white">
796                  <span className="card__label text-base">{card.label}</span>
797                </div>
798                <div className="card__content flex flex-col relative text-white">
799                  <h3 className={`card__title font-normal text-base m-0 mb-1 ${textAutoHide ? 'text-clamp-1' : ''}`}>
800                    {card.title}
801                  </h3>
802                  <p className={`card__description text-xs leading-5 opacity-90 ${textAutoHide ? 'text-clamp-2' : ''}`}>
803                    {card.description}
804                  </p>
805                </div>
806              </div>
807            );
808          })}
809        </div>
810      </BentoCardGrid>
811    </>
812  );
813};
