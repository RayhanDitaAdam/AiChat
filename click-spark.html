Props

PropertyTypeDefaultDescription
sparkColor

string

'#f00'

Color of each spark line.

sparkSize

number

—

Initial length of each spark line.

sparkRadius

number

—

How far sparks travel from the click center.

sparkCount

number

—

Number of spark lines that appear on each click.

duration

number

—

Animation duration in milliseconds.

easing

string

'ease-out'

Easing function used for the spark animation.

extraScale

number

—

Additional multiplier for spark distance.

children

React.ReactNode

—

React children to render.

code

1import { useRef, useEffect, useCallback } from 'react';
2
3const ClickSpark = ({
4  sparkColor = '#fff',
5  sparkSize = 10,
6  sparkRadius = 15,
7  sparkCount = 8,
8  duration = 400,
9  easing = 'ease-out',
10  extraScale = 1.0,
11  children
12}) => {
13  const canvasRef = useRef(null);
14  const sparksRef = useRef([]);
15  const startTimeRef = useRef(null);
16
17  useEffect(() => {
18    const canvas = canvasRef.current;
19    if (!canvas) return;
20
21    const parent = canvas.parentElement;
22    if (!parent) return;
23
24    let resizeTimeout;
25
26    const resizeCanvas = () => {
27      const { width, height } = parent.getBoundingClientRect();
28      if (canvas.width !== width || canvas.height !== height) {
29        canvas.width = width;
30        canvas.height = height;
31      }
32    };
33
34    const handleResize = () => {
35      clearTimeout(resizeTimeout);
36      resizeTimeout = setTimeout(resizeCanvas, 100);
37    };
38
39    const ro = new ResizeObserver(handleResize);
40    ro.observe(parent);
41
42    resizeCanvas();
43
44    return () => {
45      ro.disconnect();
46      clearTimeout(resizeTimeout);
47    };
48  }, []);
49
50  const easeFunc = useCallback(
51    t => {
52      switch (easing) {
53        case 'linear':
54          return t;
55        case 'ease-in':
56          return t * t;
57        case 'ease-in-out':
58          return t < 0.5 ? 2 * t * t : -1 + (4 - 2 * t) * t;
59        default:
60          return t * (2 - t);
61      }
62    },
63    [easing]
64  );
65
66  useEffect(() => {
67    const canvas = canvasRef.current;
68    if (!canvas) return;
69    const ctx = canvas.getContext('2d');
70
71    let animationId;
72
73    const draw = timestamp => {
74      if (!startTimeRef.current) {
75        startTimeRef.current = timestamp;
76      }
77      ctx.clearRect(0, 0, canvas.width, canvas.height);
78
79      sparksRef.current = sparksRef.current.filter(spark => {
80        const elapsed = timestamp - spark.startTime;
81        if (elapsed >= duration) {
82          return false;
83        }
84
85        const progress = elapsed / duration;
86        const eased = easeFunc(progress);
87
88        const distance = eased * sparkRadius * extraScale;
89        const lineLength = sparkSize * (1 - eased);
90
91        const x1 = spark.x + distance * Math.cos(spark.angle);
92        const y1 = spark.y + distance * Math.sin(spark.angle);
93        const x2 = spark.x + (distance + lineLength) * Math.cos(spark.angle);
94        const y2 = spark.y + (distance + lineLength) * Math.sin(spark.angle);
95
96        ctx.strokeStyle = sparkColor;
97        ctx.lineWidth = 2;
98        ctx.beginPath();
99        ctx.moveTo(x1, y1);
100        ctx.lineTo(x2, y2);
101        ctx.stroke();
102
103        return true;
104      });
105
106      animationId = requestAnimationFrame(draw);
107    };
108
109    animationId = requestAnimationFrame(draw);
110
111    return () => {
112      cancelAnimationFrame(animationId);
113    };
114  }, [sparkColor, sparkSize, sparkRadius, sparkCount, duration, easeFunc, extraScale]);
115
116  const handleClick = e => {
117    const canvas = canvasRef.current;
118    if (!canvas) return;
119    const rect = canvas.getBoundingClientRect();
120    const x = e.clientX - rect.left;
121    const y = e.clientY - rect.top;
122
123    const now = performance.now();
124    const newSparks = Array.from({ length: sparkCount }, (_, i) => ({
125      x,
126      y,
127      angle: (2 * Math.PI * i) / sparkCount,
128      startTime: now
129    }));
130
131    sparksRef.current.push(...newSparks);
132  };
133
134  return (
135    <div className="relative w-full h-full" onClick={handleClick}>
136      <canvas ref={canvasRef} className="w-full h-full block absolute top-0 left-0 select-none pointer-events-none" />
137      {children}
138    </div>
139  );
140};
141
142export default ClickSpark;
143
